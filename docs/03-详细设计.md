```python
# auth/decorator.py
def require_role(*roles):
    def decorator(f):
        @functools.wraps(f)
        def wrapper(*args, **kwargs):
            user = session.get("user")
            if not user or user["role"] not in roles:
                return jsonify({"error": "权限不足",
                                 "required": list(roles),
                                 "actual": user.get("role") if user else None}), 403
            return f(*args, **kwargs)
        return wrapper
    return decorator
```

## 3.4 校验器（legal/validator.py）

```python
class FaqValidator:
    REQUIRED_FIELDS = ["legal_answer", "compliance_risk",
                       "practical_advice", "legal_basis"]
    LEGAL_BASIS_RE = re.compile(r"《.+?》.*?第.+?条")
    DUPLICATE_THRESHOLD = 0.8

    def validate(self, item: Dict) -> Dict:
        errors, warnings, similar_items = [], [], []

        # 1. 问题完整性（硬）——有问题才有答案
        if not self._is_nonempty(item.get("question")):
            errors.append("问题 question 为空，必须先有问题才能作答")

        # 2. 四段式完整性（硬）
        for field in REQUIRED_FIELDS:
            if not self._is_nonempty(item.get(field)):
                errors.append(f"字段 {field} 为空")

        # 3. 法条格式（软）
        if self._is_nonempty(item.get("legal_basis")):
            if not LEGAL_BASIS_RE.search(item["legal_basis"]):
                warnings.append('法条格式不规范，建议含《法名》和"第X条"格式')

        # 4. 标签完整性（软）
        if not item.get("tags"):
            warnings.append("未设置标签，建议至少 1 个")

        # 5. 重复校验（软）
        if self._is_nonempty(item.get("question")):
            similar = self._check_duplicate(item["question"])
            if similar:
                similar_items = similar
                warnings.append(f"与已有条目 {ids} 高度相似")

        # 6. 问题格式（软）
        q = item.get("question", "").rstrip()
        if q and not q.endswith(("？", "?")):
            warnings.append('问题未以问号结尾，建议以"？"结尾')

        return {"passed": len(errors) == 0,
                "errors": errors, "warnings": warnings,
                "similar_items": similar_items}
```

## 3.5 AI 审核引擎（核心）

### 3.5.1 算法一：KeywordMatcher（粗筛）

**原理**：对查询和每条 FAQ 抽取关键词集合（单字 + 双字 + 法务术语），计算交集大小，取 Top-20 候选。

```python
class KeywordMatcher(DataAnalyzer):
    STOP_WORDS = {"什么", "是", "的", "如何", ...}
    LEGAL_TERMS = {"合同", "担保", "诉讼", "著作权", ...}

    def extract(self, text: str) -> set:
        # 单字（去停用词）+ 双字组合 + 法务术语词典命中
        cn = [t for t in tokenize(text) if re.match(r"[\u4e00-\u9fa5]", t)]
        keywords = set()
        for ch in cn:
            if ch not in self.STOP_WORDS:
                keywords.add(ch)
        for i in range(len(cn) - 1):
            bi = cn[i] + cn[i+1]
            if bi not in self.STOP_WORDS:
                keywords.add(bi)
        for term in self.LEGAL_TERMS:
            if term in text:
                keywords.add(term)
        return keywords

    def filter(self, query, items, top_n=20):
        q_kws = self.extract(query)
        scored = []
        for item in items:
            item_text = item["question"] + " " + item.get("legal_answer", "")
            score = len(q_kws & self.extract(item_text))
            if score > 0:
                scored.append((score, item))
        scored.sort(key=lambda x: -x[0])
        return [it for _, it in scored[:top_n]]
```

**为什么用双字**：单字过于泛化（"合""同"无区分度），双字可命中"合同""担保"等术语。

### 3.5.2 算法二：SimilarityRetriever（TF-IDF 精排）

**原理**：
1. `fit()`：对所有 FAQ 构建词汇表 + IDF 表 + 文档向量
2. `rank()`：对查询向量化，与候选向量计算余弦相似度，按相似度降序返回 Top-5

```python
class SimilarityRetriever(DataAnalyzer):
    def fit(self, items):
        docs = [self._doc_text(it) for it in items]
        N = len(docs)
        df = Counter()
        for doc in docs:
            for w in set(self._tokenize(doc)):
                df[w] += 1
        self.vocab = list(df.keys())
        self.idf = {w: math.log((N+1)/(df[w]+1)) + 1.0 for w in self.vocab}
        self.doc_vectors = [self._vectorize(doc) for doc in docs]

    def _vectorize(self, doc):
        # TF = 词频/文档总词数；TF-IDF = TF * IDF
        tokens = self._tokenize(doc)
        tf = Counter(tokens)
        vec = [0.0] * len(self.vocab)
        for w, count in tf.items():
            if w in self.vocab_index:
                vec[self.vocab_index[w]] = (count/len(tokens)) * self.idf[w]
        return vec

    @staticmethod
    def _cosine(a, b, a_norm=None, b_norm=None):
        # cos(θ) = (A·B) / (|A|·|B|)
        dot = sum(x*y for x, y in zip(a, b))
        return dot / (a_norm * b_norm) if a_norm and b_norm else 0.0
```

**为什么自研而非调 sklearn**：实验要求"算法逻辑需自行实现"，自研可清晰展示 TF/IDF/余弦公式实现。

### 3.5.3 算法流程图

```
用户输入
   │
   ↓
[KeywordMatcher] 提取关键词（单字/双字/法务术语）
   │
   ↓
计算与每条 FAQ 关键词集合交集 → Top-20 候选
   │
   ├── 无候选 ──→ 返回"未检索到"
   ↓
[SimilarityRetriever] TF-IDF 向量化候选
   │
   ↓
余弦相似度排序 → Top-5
   │
   ↓
[LLM] 基于检索结果生成四段式法务意见
   │
   ├── LLM 失败 ──→ 回退规则拼装
   ↓
附加免责声明 → 返回前端
```

### 3.5.4 RAG + LLM 编排器

```python
class ReviewOrchestrator:
    def __init__(self, knowledge_base, llm_client=None):
        self.kb = knowledge_base
        self.matcher = KeywordMatcher()
        self.retriever = SimilarityRetriever()
        self.llm = llm_client
        self.refresh()

    def review(self, user_input, top_k=5, with_disclaimer=True):
        # 1. 粗筛
        candidates = self.matcher.filter(user_input, self.kb.items, top_n=20)
        if not candidates:
            return {"answer": "未检索到相关知识。", "mode": "rag", ...}
        # 2. 精排
        ranked = self._rank_with_score(user_input, candidates, top_k)
        items = [r["item"] for r in ranked]
        # 3. 生成（LLM 优先，失败回退）
        answer, mode = self._generate(user_input, items)
        return {"answer": answer + DISCLAIMER, "mode": mode, ...}

    def _generate(self, query, items):
        fallback = self._compose(items)
        if not self.llm or not self.llm.available:
            return fallback, "rag"
        try:
            answer = self._generate_with_llm(query, items)
            return answer, "rag+llm"
        except LlmError:
            return fallback, "rag"

    def _generate_with_llm(self, query, items):
        # 将 Top-K 检索结果作为上下文，约束模型只依据知识库作答
        context = "\n\n".join(f"【案例{i}】..." for i, it in enumerate(items, 1))
        messages = [
            {"role": "system", "content": "你是国有企业法务合规助手..."},
            {"role": "user", "content": f"知识库参考：\n{context}\n\n业务咨询：{query}"},
        ]
        return self.llm.chat(messages, temperature=0.3, max_tokens=1024)
```

### 3.5.5 大模型客户端（OpenAI 兼容，标准库实现）

```python
class LlmClient:
    def chat(self, messages, temperature=0.3, max_tokens=1024) -> str:
        url = f"{self.base_url}/chat/completions"
        payload = {"model": self.model, "messages": messages,
                   "temperature": temperature, "max_tokens": max_tokens}
        req = urllib.request.Request(url, data=json.dumps(payload).encode(),
                                      headers={"Authorization": f"Bearer {self.api_key}",
                                               "Content-Type": "application/json"})
        with urllib.request.urlopen(req, timeout=self.timeout) as resp:
            result = json.loads(resp.read())
        return result["choices"][0]["message"]["content"]
```

### 3.5.6 用户反馈机制

**原理**：用户对检索到的知识库条目标注"相关/不相关"，系统收集反馈数据，在后续检索排序时给予正向/负向加权，逐步优化检索准确性。

```python
# orchestrator.py 反馈相关代码
_FEEDBACK_BOOST = 0.15      # 反馈影响力系数
_FEEDBACK_FILE = "data/feedback.json"  # 反馈持久化文件

class ReviewOrchestrator:
    def __init__(self, knowledge_base, llm_client=None):
        self.feedback: Dict[str, Dict[str, int]] = {}
        self._load_feedback()
        # ...

    def _load_feedback(self):
        """加载历史反馈数据。"""
        if os.path.exists(_FEEDBACK_FILE):
            with open(_FEEDBACK_FILE, "r", encoding="utf-8") as f:
                self.feedback = json.load(f)

    def _save_feedback(self):
        """保存反馈数据到磁盘。"""
        with open(_FEEDBACK_FILE, "w", encoding="utf-8") as f:
            json.dump(self.feedback, f, ensure_ascii=False)

    def record_feedback(self, faq_id: str, relevant: bool):
        """记录用户反馈。

        Args:
            faq_id: 知识库条目 ID
            relevant: True=相关, False=不相关
        """
        if faq_id not in self.feedback:
            self.feedback[faq_id] = {"positive": 0, "negative": 0}
        if relevant:
            self.feedback[faq_id]["positive"] += 1
        else:
            self.feedback[faq_id]["negative"] += 1
        self._save_feedback()

    def _rank_with_score(self, query, candidates, top_k):
        """检索排序时融入反馈加权。"""
        ranked = self.retriever.search(query, candidates, top_k)
        for item in ranked:
            faq_id = item["item"].get("id")
            if faq_id in self.feedback:
                fb = self.feedback[faq_id]
                ratio = fb["positive"] / (fb["positive"] + fb["negative"] + 1)
                item["score"] += ratio * _FEEDBACK_BOOST
        return ranked
```

## 3.8 合同审核模块（file_review，独立模块）

### 3.8.1 设计目标

模块移植自外部项目 [Contract-Review-System](https://github.com/tidu2020/Contract-Review-System)，作为独立子系统嵌入主系统。设计原则：

- **完全独立**：自带 `LLMClient`、`AppConfig`，不依赖 `ai`/`knowledge`/`business` 等模块

- **六阶段流水线**：提取 → 检索 → 业务审查 → 法务审查 → 法条核验 → 报告生成

**特殊处理**：
- LRU 缓存：相同输入命中缓存直接返回，节省调用成本
- JSON 修复：LLM 偶尔输出格式错误的 JSON，自动修复（去 ` ```json ` 包裹、修尾随逗号、单引号转双引号）

- 重试机制：可重试错误（429/超时/5xx）自动重试 2 次

### 3.8.3 六阶段流水线

```python
# file_review/src/pipeline.py
class ContractReviewPipeline:
    def __init__(self, config: Optional[AppConfig] = None,
                 step_callback: Optional[Callable] = None):
        self.config = config or AppConfig()
        self.llm = LLMClient(self.config.llm)
        self.extractor = ContractExtractor(self.llm)
        self.case_matcher = CaseCauseMatcher(self.llm, self.config.data_dir)
        self.searcher = LegalSearcher(self.llm, self.config.search)
        self.reviewer = ContractReviewer(self.llm)
        self.report_gen = ReportGenerator()
        self.step_callback = step_callback

    def run(self, contract_text, special_focus="",
            user_stance="auto", contract_context="") -> str:
        # 阶段 1：合同信息提取（合同类型、当事人、标的、价款等）
        contract_info = self.extractor.extract(contract_text)
        # 阶段 2：知识检索（基于合同类型匹配案由 + 检索法律依据）
        knowledge_base, search_kw, _ = self._build_knowledge_base(
            contract_info.contract_type, case_cause)
        # 阶段 3：业务审查（商务条款、付款方式、违约责任）
        biz_ok, *biz_results = self._safe_business_review(...)
        # 阶段 4：法务审查（法律效力、合规风险）
        legal_results = self._safe_legal_review(...)
        # 阶段 5：法条核验（高风险法条 LLM 复核）
        legal_refs = self._verify_laws(...)
        # 阶段 6：报告生成（Markdown 格式）
        return self.report_gen.generate_markdown(report)
```

**长合同分块策略**：当 `estimate_tokens(contract_text) > context_limit * 0.75 * 0.85` 时，按"第X条"切分为多个 chunk，分别审查后合并结果。
